\begin{frame}[fragile]
\frametitle{How It Works}
\begin{enumerate}
    \item Analyze original song's syllable structure
    \item Extract theme keywords and definitions
    \item Generate each line matching syllable count
    \item Maintain rhythm and style
\end{enumerate}

\begin{lstlisting}
let parody = try await Yankovinator.generateParody(
    originalLyrics: lyrics,
    keywords: keywords
)
\end{lstlisting}
\end{frame}

\begin{frame}
\frametitle{Syllable Counting}
\begin{itemize}
    \item Uses NaturalLanguage tokenization
    \item Vowel counting heuristics
    \item Special case handling:
    \begin{itemize}
        \item Silent 'e' endings
        \item 'le' endings
        \item Consonant clusters
    \end{itemize}
    \item Per-line and per-word analysis
    \item Proper handling of contractions (don't, can't, it's)
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{API Reference}
\begin{itemize}
    \item \texttt{Yankovinator.generateParody()} - Main entry point
    \item \texttt{SyllableCounter.countSyllables()} - Word analysis
    \item \texttt{SyllableCounter.analyzeSongStructure()} - Song analysis
    \item \texttt{ParodyGenerator.generateParody()} - Full generation
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Algorithm Flow}
\begin{enumerate}
    \item Parse input lyrics
    \item Analyze syllable structure
    \item Load theme keywords
    \item For each line:
    \begin{itemize}
        \item Build prompt with constraints
        \item Request from Ollama
        \item Validate syllable count
        \item Clean and format
    \end{itemize}
    \item Return complete parody
\end{enumerate}
\end{frame}
